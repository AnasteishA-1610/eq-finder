function full_syst(x, params, t)
    @inbounds fi1, V1, fi2, V2 = x[1], x[2], x[3], x[4]
    Gamma = params[1]; Lambda = params[2]; paramK = params[3]
    dx1 = V1
    dx2 = Gamma - Lambda * V1 - sin(fi1) + paramK * sin(fi2 - fi1)
    dx3 = V2
    dx4 = Gamma - Lambda * V2 - sin(fi2) + paramK * sin(fi1 - fi2)
    return SVector{4}(dx1, dx2, dx3, dx4)
end

function Lyap(del_2)
    del_1 = LinRange(-0.1, 0.1, 11)
    u0 = [0.812618351, -2.36648568e-08, 2.48961477, 5.51664837e-06]
    params = (0.6664278241282653, 0.55, 0.06)
    tspan = (0.0,2000.0)
    fig = figure(figsize=(10,40));
    count=1
    for i in del_1
        new_u0 = [u0[1] + i, 0, u0[3] + del_2, 0]
        ds = ContinuousDynamicalSystem(full_syst, new_u0, params)
        lyap = lyapunovspectrum(ds, 100000, Δt = 0.1, Ttr = 10.0)
        print(count, "\n")
        if (lyap[1]>0)
            print("positive", "\n")
            print("StartPt = ", new_u0, '\n')
            print("del_1 = $i, del_2 = $del_2", "\n")
            print("Lyapunov spectrum = ", lyap, '\n', '\n')
            prob = ODEProblem(full_syst, new_u0, tspan, params)
            sol = solve(prob, DP8(), abstol=1e-11, reltol=1e-11, dtmax = 1e-1)
            subplot(11,1,count)
            plot(sol[1,:], sol[2,:])
            show()
            title("del_1 = $i, del_2 = $del_2")
        end
        count=count+1
    end
    print("##########", "\n")
# show()
end
